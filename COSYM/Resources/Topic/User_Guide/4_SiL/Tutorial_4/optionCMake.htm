# common and overall compiler options (optimization, warning, debugging, ...)
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    set (EX_CFLAGS)
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "MSVC")
    set (EX_CFLAGS)
endif()
 
# C++ language compiler options https://cmake.org/cmake/help/latest/prop_gbl/CMAKE_CXX_KNOWN_FEATURES.html
set (EX_CPPFLAGS)
 
# C language compiler options only https://cmake.org/cmake/help/latest/prop_gbl/CMAKE_C_KNOWN_FEATURES.html#prop_gbl:CMAKE_C_KNOWN_FEATURES
set (EX_CONLYFLAGS)
 
# linker options
set (EX_LDOPTIONS)
 
# additional source files to be linked
set (EX_SRCS)
 
# additional object files to be linked
set (EX_OBJS)
 
# additional libraries to be linked
set (EX_LIBS vNetInterface.lib)
 
# additional library paths to be linked
set (EX_LIBS_PATH "$ENV{COSYM_HOME}\\Simulator\\win64\\esse\\active\\link64\\")
# additional header file paths to be linked
set (EX_INC_PATH "$ENV{COSYM_HOME}\\Simulator\\win64\\esse\\active\\include\\")
 
# additional Macros to be linked
set (EX_MACROS )
 
# linker options that have to be specified after all other options
set (EX_LDOPTIONS_END)
